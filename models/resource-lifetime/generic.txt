# Generic lifecycle rules:
#   acquire_out <function-pattern> <out-arg-index> <resource-kind>
#   acquire_ret <function-pattern> <resource-kind>
#   release_arg <function-pattern> <arg-index> <resource-kind>

# Generic out-param handle pattern
acquire_out acquire_handle 0 GenericHandle
acquire_ret acquire_handle GenericHandle
release_arg release_handle 0 GenericHandle

# C heap allocation
acquire_ret malloc HeapAlloc
release_arg free 0 HeapAlloc

# C++ ABI demangler allocates a heap buffer (freed with free()).
acquire_ret *__cxa_demangle* HeapAlloc

# C++ heap allocation (demangled names contain spaces, so use glob patterns)
acquire_ret operator*new* CppHeap
release_arg operator*delete* 0 CppHeap

# Vulkan resource lifecycle rules
# Covers vkCreate*/vkDestroy* and vkAllocate*/vkFree* pairs

# Instance & Device

# vkCreateInstance(pCreateInfo, pAllocator, pInstance*)
acquire_out vkCreateInstance 2 VkInstance
release_arg vkDestroyInstance 0 VkInstance

# vkCreateDevice(physicalDevice, pCreateInfo, pAllocator, pDevice*)
acquire_out vkCreateDevice 3 VkDevice
release_arg vkDestroyDevice 0 VkDevice

# Swapchain

# vkCreateSwapchainKHR(device, pCreateInfo, pAllocator, pSwapchain*)
acquire_out vkCreateSwapchainKHR 3 VkSwapchainKHR
release_arg vkDestroySwapchainKHR 1 VkSwapchainKHR

# Command Pool & Buffers

# vkCreateCommandPool(device, pCreateInfo, pAllocator, pCommandPool*)
acquire_out vkCreateCommandPool 3 VkCommandPool
release_arg vkDestroyCommandPool 1 VkCommandPool

# vkAllocateCommandBuffers(device, pAllocateInfo, pCommandBuffers*)
acquire_out vkAllocateCommandBuffers 2 VkCommandBuffer
# vkFreeCommandBuffers(device, commandPool, count, pCommandBuffers*)
release_arg vkFreeCommandBuffers 3 VkCommandBuffer
# Wrapper-style helpers returning a temporary command buffer consumed by an end helper.
acquire_ret *beginSingleTimeCommands* VkCommandBuffer
release_arg *endSingleTimeCommands* 3 VkCommandBuffer
